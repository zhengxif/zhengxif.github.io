# events(事件触发器)

const util = require('util');
const EventEmitter = require('events');

function MyEmitter() {

}
// 继承
// MyEmitter.prototype.__proto__ = EventEmitter.prototype; // ie不支持
// Object.setPrototypeOf(MyEmitter.prototype, EventEmitter.prototype);
util.inherits(MyEmitter,EventEmitter); 
const myEmitter = new MyEmitter;

myEmitter.on('newListener', function(type) {
    process.nextTick(() => {
        myEmitter.emit(type, '0');
    })
})
myEmitter.on('哈哈', function(num) {
    console.log('触发了哈哈' + num);
})
myEmitter.on('哈哈', function(num) {
    console.log('触发了哈哈2' + num);
})
