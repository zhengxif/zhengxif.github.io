const http = require('http');
const querystring = require('querystring');
let port = 4001;
let server = http.createServer(function(req, res) {
    let contentType = req.headers['content-type'];
    res.setHeader('Content-type', 'application/json;charset=utf-8');
    let arr = [];
    req.on('data', function(chunk) {
        arr.push(chunk)
    })
    req.on('end', function() {
        let str = Buffer.concat(arr).toString();
        if (contentType == 'application/json') {
            let obj = JSON.parse(str);
            res.end(obj.name);
        } else if (contentType == 'application/x-www-form-urlencoded') {
            let obj = querystring.parse(str, '&', '=');
            res.end(obj.name);
        } else {
            res.end('not support');
        }
    })
})

server.listen(port, function() {
    console.log('server start:' + port);
})

mime模块处理请求文件类型

const http = require('http');
const path = require('path');
const fs = require('fs');
const mime = require('mime');
const url = require('url');

http.createServer(function(req, res) {
    let { pathname } = url.parse(req.url);
    console.log(pathname);
    let absPath = path.join(__dirname, 'public', pathname);
    if (pathname == '/favicon.ico') {
        res.setHeader('Content-type', 'text/plain;charset=utf-8');
        res.end('没有 favicon.ico');
        return;
    }
    fs.stat(absPath, function(err, statObj) {
        res.setHeader('Content-type', 'text/html;charset=utf-8');
        if (err) {
            return res.end('Not found');
        }
        if (statObj.isDirectory()) {
            absPath = path.join(absPath, 'index.html'); // 如果是目录就访问index.html
            fs.stat(absPath, function(err, statObj) {
                if (err) {
                    return res.end('Not found');
                }
                res.setHeader('Content-type', 'text/html;charset=utf-8');
                fs.createReadStream(absPath).pipe(res);
            })
        } else {
            let type = mime.getType(absPath);
            if (!type) {
                res.setHeader('Content-Disposition', `attachment;filename=${encodeURIComponent('下载')+path.extname(absPath)}`);
            }
            res.setHeader('Content-type', `${type};charset=utf-8`);
            fs.createReadStream(absPath).pipe(res); // 会默认调用可写流的write和end方法
        }
    })
}).listen(4003);

# class+async版静态服务

const http = require('http');
const path = require('path');
const mime = require('mime');
const url = require('url');
let fs = require('fs').promises;
let {createReadStream} = require('fs');

class Server{
    async handleServer(req, res) {
        let { pathname } = url.parse(req.url);
        let absFilePath = path.join(__dirname, pathname);
        try {
            let statObj = await fs.stat(absFilePath);
            if (statObj.isDirectory()) {
                absFilePath = path.join(absFilePath, 'index.html');
            }
            this.sendFile(req, res, absFilePath, statObj)
        } catch (error) {
            console.log(error);
            this.sendError(req, res);
        }

    }
    sendFile(req, res, absFilePath, statObj) {
        let type = mime.getType(absFilePath);
        console.log(type)
        res.setHeader('Content-type', `${type};charset=utf-8;`);
        createReadStream(absFilePath).pipe(res);
    }
    sendError(req, res) {
        res.end('Not found');
    }
    start() {
        let server = http.createServer(this.handleServer.bind(this));
        server.listen(...arguments);
    }
}

let server = new Server();

# 静态服务+强缓存+协商缓存

const path = require('path');
const http = require('http');
const url = require('url');
const fs = require('fs').promises;
const { createReadStream, readFileSync } = require('fs');
const mime = require('mime');
const nunjucks = require('nunjucks');
const chalk = require('chalk');
const crypto = require('crypto');
class Server {
    constructor(config) {
        this.port = config.port;
        this.dir = config.dir;
    }
    async serverHandler(req, res) {
        let { pathname } = url.parse(req.url);
        pathname = decodeURIComponent(pathname);
        if(pathname === '/favicon.ico') return res.end(`Not Found`);
        let absPath = path.join(this.dir, pathname);
        try {
            let statObj = await fs.stat(absPath);
            if (statObj.isFile()) {
                this.sendFile(absPath, req, res, statObj);
            } else {
                // 需要列出所有文件夹内容
                let children = await fs.readdir(absPath);
                children = children.map(item => ({
                    current: item,
                    parent: path.join(pathname, item)
                }));
                let str = nunjucks.render(path.resolve(__dirname, 'template.html'), { data: children });
                res.setHeader('Content-type', 'text/html;charset=utf-8');
                res.end(str);
            }
        } catch (error) {
            console.log(error);
            this.sendError(error, res)
        }
    }
    hasCache(absPath, req, res, statObj) {
        // 强缓存10毫秒
        res.setHeader('Cache-Control', 'max-age=10'); 
        res.setHeader('Expires', new Date(Date.now() + 10*1000).toGMTString());
        
        // 协商缓存
        let ctime = statObj.ctime.toGMTString(); // 拿到文件变化的时间
        res.setHeader('Last-Modified', ctime);
        let content = readFileSync(absPath, 'utf8');
        let etag = crypto.createHash('md5').update(content).digest('base64');
        res.setHeader('Etag', etag);

        //第二次访问， 取值对比
        let ifModifiedSince = req.headers['if-modified-since'];
        if (ifModifiedSince != ctime) {
            return false;
        }
        let ifNoneMatch = req.headers['if-none-match'];
        if (ifNoneMatch != etag) {
            return false
        }
        return true;

    }
    sendFile(absPath, req, res, statObj) {
        if(this.hasCache(absPath, req, res, statObj)) {
            res.statusCode = 304;
            return res.end();
        }
        res.setHeader('Content-type', mime.getType(absPath) + ";charset=utf-8");
        createReadStream(absPath).pipe(res);
    }
    sendError(err, res) {
        res.statusCode = 404;
        res.end('Not found');
    }
    start() {
        let server = http.createServer(this.serverHandler.bind(this));
        server.listen(this.port, () => {
            console.log(`${chalk.yellow('Starting up http-server, serving') }./${this.dir.split('/').pop()}
    Available on:
    http://127.0.0.1:${chalk.green(this.port)}
    Hit CTRL-C to stop the server`);
        })
    }
}

module.exports = Server;

const http = require('http');
const fs = require('fs').promises;
const path = require('path');
const url = require('url');

http.createServer(async function(req, res) {
    let { pathname, query } = url.parse(req.url, true);
    let absPath = path.resolve(__dirname,'.', pathname);
    // 1) 判断是否是接口
    try {
        let statObj = await fs.stat(absPath);
        if (statObj.isFile) { //只对文件做处理
            let content = await fs.readFile(absPath);
            res.end(content);
        }
    } catch (error) {
        // 2) 可能是api接口
        let type = req.headers['content-type'];
        let arr = [];
        req.on('data', function(chunk) {
            arr.push(chunk);
        })
        req.on('end', function() {
            // 允许跨域
            if (req.headers.origin) { // 针对跨域做处理
                res.setHeader('Access-Control-Allow-Origin', req.headers.origin); // 允许所有的域名访问
                res.setHeader('Access-Control-Allow-Credentials', true); // 允许携带cookie凭证
                res.setHeader('Access-Control-Allow-Headers', 'token,a,b'); // 允许携带哪些头
                res.setHeader('Access-Control-Max-Age', '10'); // 最大存活时间， 预检请求的时间循环
                res.setHeader('Access-Control-Allow-Methods', 'PUT,DELETE,OPTIONS'); // 允许复杂请求
                // 如果是options直接断开
                if (req.method === 'OPTIONS') {
                    return res.end()
                }
                if (pathname == '/user') { // restful风格
                    switch (req.method) {
                        case 'GET':
                            res.setHeader('Content-Type', 'application/json');
                            return res.end(JSON.stringify({
                                name: 'zx'
                            }))
                        case 'POST':
                            if (type == 'application/x-www-form-urlencoded') {
                                let str = Buffer.concat(arr).toString();
                                let data = require('querystring').parse(str);
                                res.setHeader('Content-type', 'application/json');
                                res.end(JSON.stringify(data));
                            }
                        case 'DELETE':
                            res.setHeader('Content-Type','application/json');
                            res.setHeader('Set-Cookie', 'name=zx');
                            return res.end(JSON.stringify({
                                name: 'zx'
                            }))
                        default:
                            break;
                    }
                }   
            }
            console.log(pathname, query.callback)
            if (pathname == '/jsonp') {
                return res.end(`${query.callback}('a=1')`);
            }
            res.statusCode = 404;
            res.end('Not found');
        })
    }
}).listen(4000);

const path = require('path');
const fs = require('fs').promises;
const url = require('url');
const http = require('http');

http.createServer(async function (req, res) {
    let { pathname } = url.parse(req.url);
    let absPath = path.join(__dirname, pathname);
    try {
        let statObj = await fs.stat(absPath);
        if (statObj.isFile()) {
            if (pathname.match(/\.jpg/)) { // 对访问图片做防盗链
                let referer = req.headers['referer'] || req.headers['referrer'];
                console.log(referer);
                if (referer) {
                    // 根据判断host referer
                    referer = url.parse(referer).hostname;
                    let host = req.headers['host'].split(':')[0];
                    if(host != referer) {
                        let content = await fs.readFile(path.resolve(__dirname, '2.jpg'));
                        return res.end(content);
                    }
                }
            }
            let content = await fs.readFile(absPath);
            res.end(content);
        }else {
            res.statusCode = 404;
            res.end('Not found');
        }
    } catch (error) {
        console.log(error);
        res.statusCode = 404;
        res.end('Not found');
    }
}).listen(4000);

const path = require('path');
const fs = require('fs');
const http = require('http');
const url = require('url');
let defaultLanguage = 'hello';
let languages = {
    'zh-CN':{
        message: '你好'
    },
    ja:{
        message:"こんにちは"
    }
}
http.createServer(async function(req, res) {
    let { pathname } = url.parse(req.url);
    pathname = decodeURIComponent(pathname);
    console.log(pathname)
    let absPath = path.join(__dirname, pathname);
    try {
        let statObj = await fs.stat(absPath);
        if (statObj.isFile()) {
            let content = await fs.readFile(absPath);
            res.end(content);
        } else {
            res.statusCode = 404;
            res.end('Not found');
        }
    } catch (error) {
        if (pathname = '/encoding') {
            let language = req.headers['accept-language'];
            res.setHeader('Access-Control-Allow-Origin', '*');
            // zh-CN zh;q=0.9 en;q=0.7
            if (language) {
                let langs = language.split(',').map(item => {
                    let [name, q] = item.split(';');
                    return {
                        name,
                        q: q ? q.split('=')[1] : 1
                    }
                }).sort((a,b) => b.q-a.q)
                for (i = 0; i < langs.length; i++) {
                    if (languages[langs[i].name]) {
                        res.end(languages[langs[i].name].message);
                        return;
                    }
                }
                res.end(defaultLanguage);
            } else {
                res.end(defaultLanguage);
            }
        }
        res.statusCode = 404;
        res.end('Not found');
    }
}).listen(4000)

# gzip压缩

const http = require('http');
const fs = require('fs');
const zlib = require('zlib');
// Accept-Encoding: gzip, deflate, br 客户端自己发送的
// Content-Encoding: gzip
http.createServer(function(req, res) {
    if (req.url == '/index.html') {
        // 判断是否支持gzip压缩
        let headers = req.headers['accept-encoding'];
        if (headers.match(/\bgzip\b/)) {
            res.setHeader('Content-Encoding','gzip');
            res.setHeader('Content-Type','text/html;charset=utf-8');
            fs.createReadStream('./index.html').pipe(zlib.createGzip()).pipe(res);
        } else if (headers.match(/\bdeflate\b/)) {
            fs.createReadStream('./index.html').pipe(zlib.createDeflate()).pipe(res);
        } else {
            fs.createReadStream('./index.html').pipe(res);
        }
    } else {
        res.statusCode = 404;
        res.end('Not found');
    }
}).listen(4000)

# 代理 http-proxy

const http = require('http');
const httpProxy = require('http-proxy');
const proxy = httpProxy.createProxyServer();
let map = {
    'zx1.cn': 'http://localhost:3000', // 访问到zx1.cn:9000 代理到http://localhost:3000
    'zx2.cn': 'http://localhost:4000', // 访问到zx2.cn:9000 代理到http://localhost:4000
}
http.createServer(function(req, res) {
    proxy.on('proxyRes', function(proxyRes, req, res) {
        let body = [];
        res.setHeader('Content-Type', 'text/plain;charset=utf-8');
        proxyRes.on('data', function(chunk) {
            body.push(chunk);
        })
        proxyRes.on('end', function() {
            body = Buffer.concat(body).toString();
            console.log("来自代理的响应:", body);
            res.end("原来的响应");
        })
    })
    proxy.web(req, res, {
        selfHandleResponse: true,
        target: map[req.headers['host'].split(':')[0]]
    })
}).listen(9000)

# 断点续传 206

// 写一个客户端 不停的更改 start 和 end
const http = require('http');
const fs = require('fs');
let start = 0;
let size = 3;
let flowing = true;
process.stdin.on('data', function(data) {
    if (data.toString().includes('stop')) {
        flowing = false;
    } else {
        flowing = true;
        download();
    }
})
function download() {
    http.get({
        host: 'localhost',
        port: 4000,
        headers: {
            Range: `bytes=${start}-${start+size}`
        }
    }, function(res) {
        setTimeout(function() {
            res.on('data', function(chunk) {
                fs.appendFileSync('./2.txt', chunk);
                start += chunk.length;
                let total =  res.headers['content-range'].split('/')[1];
                if (start < total && flowing) {
                    download();
                }
            })
        }, 1000)
    })
}
download();

// 服务端
const http = require('http');
const fs = require('fs');
let size = fs.statSync('./1.txt').size;

http.createServer(function(req, res) {
    let range = req.headers['range'];
    if (range) {
        let [, start, end] = range.match(/(\d+)-(\d+)/);
        start = Number(start);
        end = Number(end);
        res.statusCode = 206;
        res.setHeader('Content-Range',`bytes ${start}-${end}/${size}`);
        fs.createReadStream('./1.txt', { start, end }).pipe(res);
    } else {
        fs.createReadStream('./1.txt').pipe(res);
    }
}).listen(4000)

# cookie/session

  • 存储的位置 cookie 存在浏览器的 而且浏览器可以篡改 可以做签名 cookie每次请求都会带上 浪费流量 4k
  • session 基于cookie 是存在于服务器的 相对于cookie是安全的, 重启服务器消失,一般存在redis中
  • localStorage 存在本地 不能跨域访问 默认不销毁 需要手动清除 5m
  • sessionStroage 浏览器关掉会话销毁

# cookie + signed签名

const http = require('http');
const querystring = require('querystring');
const crypto = require('crypto');
const secret = 'zx0229'; // 服务端自己设置的密钥
const sign = function(value) {
    return crypto.createHmac('sha256', secret).update(value).digest('base64').replace(/\=/g, '').replace(/\+/g,'-').replace(/\//g,'_');
}
http.createServer(function(req, res) {
    res.getCookie = function(key, options = {}) {
        let obj = querystring.parse(req.headers['cookie'], '; ');
        let cookie = obj[key];
        let [value, s] = cookie.includes('.')? cookie.split('.') : [cookie];
        if (options.signed) {
            if (sign(value) === s) { // 校验cookie的签名，如果没有更改说明值是服务端设置的
                return value;
            } else {
                return '';
            }
        }
        return obj[key];
    }
    let cookies = [];
    res.setCookie = function(key, value, options = {}) {
        let arr = [];
        let cookie = `${key}=${value}`;

# session

  • cookie 是不安全的 如果服务端设置的cookie 浏览器可以看到具体内容
  • session 来基于 cookie实现的安全些
const http = require('http');
const querystring = require('querystring');
const crypto = require('crypto');
const uuid = require('uuid');
const secret = 'zx0229'; // 服务端设置签名的密钥
const cardName = 'kdzs'; // 服务端设置session时给客户端设置的cookie名, 下次访问服务端带上
let session = {}; // 一般存在redis缓存数据库
const sign = function(value) {
    return crypto.createHmac('sha256', secret).update(value).digest('base64').replace(/\=/g, '').replace(/\+/g,'-').replace(/\//g,'_');
}
http.createServer(function(req, res) {
    res.getCookie = function(key, options = {}) {
        let obj = querystring.parse(req.headers['cookie'], '; ');
        let cookie = obj[key];
        if (cookie) {
            let [value, s] = cookie.includes('.')? cookie.split('.') : [cookie];
            if (options.signed) {
                if (sign(value) === s) { // 校验cookie的签名，如果没有更改说明值是服务端设置的
                    return value;
                } else {
                    return '';
                }
            }
        }
        return obj[key];
    }
    let cookies = [];
    res.setCookie = function(key, value, options = {}) {
        let arr = [];
        let cookie = `${key}=${value}`;
        if (options.httpOnly) {
            arr.push('httpOnly=true');
        }
        if (options.maxAge) {
            arr.push(`max-age=${options.maxAge}`);
        }
        if (options.signed) {
            cookie = cookie + '.' + sign(value);
        }
        cookies.push(`${cookie}; ${arr.join('; ')}`);
        res.setHeader('Set-Cookie', cookies);
    }
    if (req.url === '/visit') {
        let cardId = res.getCookie(cardName, {
            signed: true
        })
        console.log(session);

        if (cardId && session[cardId]) {
            session[cardId].visit ++;
            res.setHeader('Content-Type', 'text/html;charset=utf-8');
            res.end(`欢迎${session[cardId].visit}次访问我`);
        } else { // 表示第一次访问
            let cardId = uuid.v4();
            res.setCookie(cardName, cardId, {
                signed: true
            });
            session[cardId] = {
                visit: 1
            }
            res.setHeader('Content-Type', 'text/html;charset=utf-8')
            res.end(`欢迎第一次访问我`)
        }
    }

}).listen(4000)
