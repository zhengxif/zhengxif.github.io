# stream(流)

一共有四种流：可读流、可写流、双工流、转化流

fs 模块下 直接可以使用stream模块

  • 实现可读流
const { Readable } = require('stream');
// 先创建一个类 这个类要继承Readable接口,内部默认会调用read方法
// 还要在提供一个_read方法
class MyReadable extends Readable {
    constructor() {
        super();
        this.index = 0;
    }
    _read() {
        this.push(this.index++ + '');
        if (this.index === 10) {
            this.push(null);
        }
    }
}
let mr = new MyReadable();
let arr = [];
mr.on('data', function(chunk) {
    arr.push(chunk);
})
mr.on('end', function() {
    console.log(Buffer.concat(arr).toString());
})
  • 实现可写流
const { Writable } = require('stream');
// Writable 接口  write方法，会默认调用子类的_write方法
class MyWritable extends Writable{
    constructor() {
        super();
    }
    _write(chunk, encoding, callback) {
        console.log(chunk.toString());
        callback();
    }
}
let mw = new MyWritable;
mw.write('你好');
mw.write('你好1');
  • 实现双工流 同时实现_read和_write方法
const { Writable, Readable } = require('stream');

class MyDuplex extends Duplex {
    constructor() {
        super();
    }
    _read() {
        this.push('双工流');
        this.push(null);
    }
    _write(chunk, encoding, cb) {
        console.log(chunk.toString());
        cb();
    }
}
let md = new MyDuplex;
md.on('data', function (chunk) {
    console.log(chunk.toString());
})
md.write('ok');
md.write('ok1');
  • 实现转化流
const { Transform } = require('stream');

class MyTransform extends Transform{
    constructor() {
        super()
    }
    _transform(chunk, encoding, cb) {
        chunk = chunk.toString().toUpperCase();
        this.push(chunk);
        cb();
    }
}
let mt = new MyTransform;
process.stdin.pipe(mt).pipe(process.stdout)
