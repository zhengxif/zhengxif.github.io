# application

const Emitter = require('events');
const http = require('http');
const context = require('./context');
const request = require('./request');
const response = require('./response');
const stream = require('stream');

module.exports = class Application extends Emitter{
    constructor() {
        super();
        this.context = Object.create(context);
        this.request = Object.create(request);
        this.response = Object.create(response);
        this.middlewares = [];
    }
    use(fn) {
        this.middleWares.push(fn);
    }
    createContext(req, res) {
        let ctx = Object.create(this.context);
        ctx.request = Object.create(this.request);
        ctx.response = Object.create(this.response);
        ctx.req = ctx.request.req = req;
        ctx.res = ctx.response.res = res;
        return ctx;
    }
    compose(middlewares, ctx) {
        function dispatch(index){
            if (index === middlewares.length) return Promise.resolve();
            try {
                return Promise.resolve(middlewares[index](ctx, () => dispatch(index+1)));
            } catch (error) {
                return Promise.reject(error);
            }
        }
        return dispatch(0);
    }
    requestHandle(req, res) {
        let ctx = this.createContext(req, res);
        this.compose(this.middlewares, ctx).then(() => {
            let body = ctx.body;
            if (body instanceof Stream) {
                body.pipe(res);
            } else if(typeof body === 'Object') {
                res.set('Content-Type', 'application/json;charset=utf-8');
                res.end(JSON.parse(body));
            } else if (typeof body === 'string' || Buffer.isBuffer(body)) {
                res.end(body);
            }else {
                res.end('Not Found');
            }
        }, err => {
            this.emit('error', err);
            res.statusCode = 500;
            res.end('server error')
        })
        
    }
    listen() {
        let app = http.createServer(this.requestHandle.bind(this));
        app.listen(4000);
    }   
}

const querystring = require('querystring');

module.exports = () => {
    return async (ctx, next) => {
        // 先把请求体解析出来
        await new Promise((resolve, reject) => {
            if (ctx.request.method == 'POST') {
                let arr = [];
                ctx.req.on('data', function(chunk) {
                    arr.push(chunk);
                })
                ctx.req.on('end', function() {
                    body = querystring.parse(Buffer.concat(arr).toString(), '&', '=');

                    resolve();
                })
            } else {
                resolve();
            }
        })
        await next();
    }
}

# 原理

const path = require('path');
const fs = require('fs').promises;

module.exports = () => {
    return async (ctx, next) => {
        let pathname = ctx.path;
        let absPath = path.join(__dirname, pathname);
        try {
            let statObj = await fs.stat(absPath);
            if (statObj.isFile()) {
                ctx.body = await fs.readFile(absPath, 'utf-8');
            } else if (statObj.isDirectory()){ 
                ctx.body = 'Not Found';
            }
        } catch (error) { // 文件不存在
            console.log(error);
            ctx.body = 'Not Found';
            await next();
        }
    }
}

# upload上传文件解析

 <form action="/form" method="POST" enctype="multipart/form-data">
    用户名<input type="text" name="username" />
    密码<input type="text" name="password" />
    <input type="file" name="pic" />
    <button type="submit">提交</button>
</form>

const Koa = require('koa');
const path = require('path');
const static = require('koa-static');
const app = new Koa();
const uuid = require('uuid');
const fs = require('fs').promises;

app.use(static(path.resolve(__dirname)));
Buffer.prototype.split = function(sep) {
    let arr = [];
    let len = Buffer.from(sep).length;
    let current;
    let offset = 0;
    while((current = this.indexOf(sep, offset)) !== -1) {
        arr.push(this.slice(offset, current));
        offset = current + len;
    }
    arr.push(this.slice(offset));
    return arr;
}
app.use(async (ctx) => {
    if (ctx.method.toLowerCase() === 'post' && ctx.path === '/form') {
        try {
            await new Promise((resolve, reject) => {
                let arr = [];
                ctx.req.on('data', function(chunk) {
                    arr.push(chunk);
                })
                ctx.req.on('end', function() {
                    let fields = {};
                    let allData = Buffer.concat(arr); 
                    // 拿到字段的分割符
                    let boundary = '--' + ctx.get('Content-Type').split('=')[1];
                    // 截取掉前后的内容
                    let lines = allData.split(boundary).slice(1, -1);
                    lines.forEach( async (line) => {
                        let [head, body] = line.split('\r\n\r\n');
                        head = head.toString();
                        let key = head.match(/name="(.+)"/)[1];
                        if (!key.includes('filename')) {
                            fields[key] = body.toString().slice(0, -2);
                        } else {
                            let filepath = path.resolve(__dirname, 'upload', uuid.v4());
                            let content = line.slice(head.length + 4, -2);
                            fields[key] = filepath;
                            await fs.writeFile(filepath, content);
                        }
                    })
                    ctx.request.fields = fields;
                    resolve();
                })
            })
        } catch (error) {
            console.log(error);
        }
        console.log(ctx.request.fields)
        ctx.body = ctx.request.fields;
    } else {
        ctx.body = 'Not found';
    }
})
app.listen(3000);

module.exports = (dirname, {map}) => {
    return async (ctx, next) => {

# koa-router

  • 一级路由使用
const Koa = require('koa');
const Router = require('koa-router');
const router = new Router();
const app = new Koa();

router.get('/a', async (ctx, next) => {
    ctx.a = 1;
    next();
})
router.get('/a', async (ctx, next) => {
    ctx.a ++;
    next();
})

router.get('/b', async (ctx, next) => {
    ctx.a ++;
    next();
})

app.use(router.routes());
app.use(ctx => {
    ctx.body = ctx.a;
})
app.listen(3000);

原理

module.exports = class {
    constructor() {
        this.middlewares = [];
    }
    get(pathname, middleware) {
        this.middlewares.push({
            pathname,
            fn:middleware
        })
    }
    compose(ctx, middlewares, next) {
        function dispatch(index) {
            if (index === middlewares.length) return next();
            return Promise.resolve(middlewares[index].fn(ctx, () => dispatch(index + 1)));
        }
        return dispatch(0);
    }
    routes() {
        return async (ctx , next) => {
            let pathname = ctx.path;
            let middlewares = this.middlewares.filter( middleware => middleware.pathname == pathname);
            this.compose(ctx, middlewares, next);
        }
    }
}

const Koa = require('koa');
const Router = require('koa-router');
const session = require('koa-session');
const router = new Router();
const app = new Koa();

app.keys = ['zx'];
app.use(session({}, app));
router.get('/write', ctx => {
    ctx.cookies.set('name', 'zx', {signed:true});
    ctx.body = 'cookie 设置成功';
})
router.get('/read', ctx => {
    ctx.body = ctx.cookies.get('name', {signed:true});
})
router.get('/visit', ctx => {
    if (ctx.session.visit) {
        ctx.session.visit ++;
    } else {
        ctx.session.visit = 1
    }
    ctx.body = `第${ctx.session.visit}访问`;
})
app.use(router.routes());
app.listen(3000);
