    # call、apply、bind

    • call
    // call 实现
// 1) 改变this指向 2) 让函数执行
function a() {
    console.log(this);
}
function b() {
    console.log(100);
}
// 函数原型上上的call方法
Function.prototype.call = function (context, ...args) {
    context = context ? Object(context) : window;
    context.fn = this; // call的原理就是将函数放到一个对象上，通过对象调用
    context.fn(...args);
    delete context.fn;
}
// 不管多少次 fn.call.call.call.call  = 原型上的call方法
// call.call(fn1)
// call方法改变this为fn1
// call函数执行，fn1执行
a.call.call.call.call.call(b); // 100
a.call(1); // 1
    • apply,和call区别就是参数是个数组
    Function.prototype.apply = function (context, args) {
    context = context ? Object(context) : window;
    context.fn = this; 
    context.fn(...args);
    delete context.fn;
}
    • bind, 内部调用call方法，改变this
    function fn1() {
    console.log(this);
}
Function.prototype.bind = function(context) {
    let that = this;
    return function() {
        console.log(this+ '-----'); // 这个是第二次bind的this,2
        that.apply(context); // 这个是第一次bind的this,1, 也就是最终执行的上下文
    }
}
let bindFn = fn1.bind(1);
let fn = bindFn.bind(2);
fn(); // 1  所以多次bind只能第一次生效
