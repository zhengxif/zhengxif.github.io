    # ES6

    以前声明变量用var，现在为什么用let、const？

    1. var会污染全局变量（常见的作用域window，function,width）
    2. var会造成变量提升
    3. var可以被多次定义
    4. var不能声明常量
    5. var默认不会产生作用域（let+{}可以产生一个作用域）

    解构赋值

    1. 解构的方式都是根据key来实现
    let [ age, name ] = [ '姓名'，'年龄' ];
    1. 用：号来重命名， = 来赋值默认值
    let { name, age:age1, address = 'zz' } = { name: '11', age:10 };

    实现深拷贝

    怎么判断数据类型

    1. typeof, 不好判断object和Array
    2. Object.prototype.toString.call(), 不好判断是谁的实例
    3. instanceof, 可以判断类型，并知道是谁的实例
    4. constructor构造函数
    const deepClone = (value, hash = new WeakMap()) => {
    if (value == null) return value; // 排除掉null,undefined
    if (typeof value != 'object') return value; //包括function
    if (value instanceof RegExp) return new RegExp(value); // 返回新的正则
    if (value instanceof Date) return new Date(value); // 返回新的日期
    // 开始拷贝，可能是数组、对象
    let instance = new value.constructor; // 根据当前属性构造新的实例
    if (hash.has(value)) {
        // 如果存在，表示已经拷贝过了，直接返回
        return hash.get(value);
    }
    hash.set(value, instance);
    for(let key in value) {
        // 过滤原型上的属性
        if (value.hasOwnProperty(key)) {
            instance[key] = deepClone(value[key], hash); // 将hash继续向下传递，保证这次拷贝能拿到以前拷贝的结果
        }
    }
    return instance;
};
let arr = [1, 2, 3, [4, 5, 6]];
let newArr = deepClone(arr);
newArr[3][2] = 100;
console.log(arr, newArr);

    实现mergeOptions

    function mergeOptions(o1, o2) {
    // 如果两个类型不相等直接返回o2即可
    if (typeof o1 !== typeof o2) {
        return o2;
    }
    if (o1.constructor !== o2.constructor) {
        return o2;
    }
    // 数组
    if (Array.isArray(o1)) {
        return [...new Set([...o1, ...o2])];
    }
    // 剩下的就是对象
    let mergeObj = {};
    let k1 = Object.keys(o1); 
    let k2 = Object.keys(o2);
    [...k1, ...k2].forEach( key => {
        mergeObj[key] = mergeOptions(o1[key], o2[key]);
    })
    return mergeObj;
}
let obj1 = {a:1,q:[1,2,3],a:{a:100,c:[1,2]},p:9};
let obj2 = {a:2,b:3,q:[4,5],a:{a:200,c:[4,5],p:9}};
let obj = mergeOptions(obj1,obj2);
console.log(obj);

    Object.defineProperty(),es5定义对象属性

        enumerable: true, // 是否可枚举，不可枚举不能被for in所迭代

let handler = {
    get(target, key) {
        if (typeof target[key] === 'object') {
            return new Proxy(target[key], handler);
        }
        // return target[key];
        return Reflect.get(target, key);
    },
    set(target, key, value) {
        // 如果是数组，长度变化不需要更新
        if (key === 'length') return true;

        // target[key] = value;

}

    es5模拟es6的类(class)

    // babel编译出来
function define(target, properties) {
    for(var i = 0; i < properties.length; i++) {
        var property = properties[i];
        Object.defineProperty(target, property.key, {
            configurable: true, // 可配置
            enumerable: false, // 不可枚举,
            value: property.value
        })
    }
}
function _ceateClass(Constructor, protoProperties, staticProperties) {
    if (Array.isArray(protoProperties)) {
        define(Constructor.prototype, protoProperties)
    }
    if (Array.isArray(staticProperties)) {
        define(Constructor, staticProperties)
    }
}
const Animal = (function () {
    function Animal() {
        if(!(this instanceof Animal)) {
            return new Error('not new')
        }
        this.name = '熊猫';
    }
    _ceateClass(
        Animal,
        [ // 原型上的方法，公共方法
            {
                key: 'say',
                value: function () {
                    console.log('say');
                }
            },
            {
                key: 'eat',
                value: function () {
                    console.log('eat');
                }
            }
        ],
        [ // 描述类上的属性或者方法
            {
                key: 'a',
                value: function () {
                    return 123;
                }
            },
            {
                key: 'b',
                value: '456'
            }

})()

    es5继承, call + Object.create = extends

    function Animal() {

Animal.prototype.say = function() {
    console.log('说话');

function Tiger() {
    // 1) 继承实例的属性
    Animal.call(this);
}
// 2) 继承原型上的公共属性
// Tiger.prototype = Animal.prototype;  这种是混用，不靠谱

// Tiger.prototype.__proto__ = Animal.prototype; //  ie 不支持直接操作__proto__;

Tiger.prototype = Object.create(Animal.prototype,{
    constructor: {
        value: Tiger
    } 
});
// 实现Object.create()方法
function create(parentPrototype) {
    function Fn() {}
    Fn.prototype = parentPrototype.prototype;
    return new Fn();
}
