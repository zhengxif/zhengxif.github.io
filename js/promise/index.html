    # Promise原理

    • promise有三个状态，等待(pengding)、成功(fulfilled)、失败(rejected)
    • promise是一个类，在new的时候是等待态
    • 每次new的时候会传入一个executor执行器，并立即执行
    • 在调用resolve的时候是成功态
    • 在调用reject的时候是失败态
    • 默认只有在等待态的时候才可以改变状态，并且状态不可逆转

    promise第一版

    // 三个常量状态
const PENGDING = 'PENGDING';
const FULFILLED = 'FULFILLED';
const REJECTED = 'REJECTED';
class Promise{
    constructor(executor) {
        this.status = PENGDING;
        this.value = undefined; // 成功的值
        this.reason = undefined; // 失败的值
        
        let resolve = value => { // 每个promise都有自己的成功和失败，不是公共的
            if (this.status === PENGDING) {
                this.value = value;
                this.status = FULFILLED;
            }
        }
        let reject = error => {
            if (this.status === PENGDING) {
                this.reason = error;
                this.status = REJECTED;
            }
        }
        try{
            // executor会立即执行
            executor(resolve, reject);
        }catch(error) {
            reject(error);
        }
    }
    then(onFulfilled, onRejected) {
        if (this.status === FULFILLED) {
            return onFulfilled(this.value);
        }
        if (this.status === REJECTED) {
            return onRejected(this.reason);
        }
    }
}
    • 每个实例都有个then方法，then方法返回一个新的promise实例
    • 如果then成功或者失败的结果，返回一个promise,会等待这个promise的执行结果，并向外层的then中传递，将结果作为参数传入
    • 如果then（成功方法、失败方法）抛出异常，则会走下一个then的失败

    注意，走失败的两种情况，返回一个失败的promise、抛出异常。

    那么如何终断then的链式调用？

    我们只需要return new Promise((resolve, reject) => {}返回一个不成功也不失败的promise;

    promise第二版

    // 三个常量状态
const PENGDING = 'PENGDING';
const FULFILLED = 'FULFILLED';
const REJECTED = 'REJECTED';
function isPromise(value) {
    if ((typeof value === 'object' && value !== null) || typeof value === 'function') {
        if (typeof value.then === 'function') {
            return true;
        }
    }
    return false;
}
function resolvePromise(promise2, x, resolve, reject) {
    // x 来取决promise2是成功还是失败
    if (promise2 === x) { // 自己等待自己完成
        return reject(new Error(
            '出错了，返回相同的promise'
        ))
    }
    // 怎么判断x是不是promise
    // 判断x是不是promise 如果x是常量，那就直接用这个结果
    if ((typeof x === 'object' && x !== null) || typeof x === 'function') {
        // {}.then x有可能定义了then方法
        // 有可能是promise
        try{
            // 可能有地方这么定义了then方法
            // Object.defineProperty(x, 'then', {
            //     get() {
            //         throw new Error();
            //     }
            // })
            // 所以try下
            let then = x.then;
            if (typeof then === 'function') {
                // 这里就只能认为是一个promise
                // x.then(y => {}, r => {});
                // 为了防止第二次x.then时抛异常，所以
                then.call(
                    x, 
                    y => {
                        // 递归解析当前x的promise的返回值，因为promise成功后可能返回的还是一个promise
                        resolvePromise(promise2, y, resolve, reject);
                    },
                    r => {
                        reject(r);
                    }
                )
            }else {
                // {}, 普通值
                resolve(x);
            }
        }catch(error) {
            reject(error);
        }
    }else {
        // 普通字符串、number、boolean
        resolve(x);
    }
}

class Promise{
    constructor(executor) {
        this.status = PENGDING;
        this.value = undefined; // 成功的值
        this.reason = undefined; // 失败的值
        this.resolveCallback = []; // 收集成功的回调
        this.rejectCallback = []; // 收集失败的回调
        let resolve = value => { // 每个promise都有自己的成功和失败，不是公共的
            if (value instanceof Promise) {
                return value.then(resolve, reject);
            }
            if (this.status === PENGDING) {
                this.value = value;
                this.status = FULFILLED;
                this.resolveCallback.forEach(fn => fn());
            }
        }
        let reject = error => {
            if (this.status === PENGDING) {
                this.reason = error;
                this.status = REJECTED;
                this.rejectCallback.forEach(fn => fn());
            }
        }
        try{
            // executor会立即执行
            executor(resolve, reject);
        }catch(error) {
            reject(error);
        }
    }
    // x 是当前then成功或者失败的返回结果
    // x 是不是普通值, 如果是普通值,就直接传递到下一个then中
    // x 如果是promise，需要采用这个promise的状态
    // x 如果执行出错，直接调用promise2的reject
    then(onFulfilled, onRejected) {
        // 两参数可选，处理下，保证传进来的是函数
        onFulfilled = typeof onFulfilled === 'function' ? onFulfilled : value => value;
        onRejected = typeof onRejected === 'function' ? onRejected : err => {
            throw err;
        } 
        let promise2 = new Promise((resolve, reject) => { // 此函数会立即执行
            if (this.status === FULFILLED) {
                setTimeout(() => {
                    try{
                        let x = onFulfilled(this.value);
                        // 看x的返回值是不是promise，在去让promise2变成成功或者失败
                        resolvePromise(promise2, x, resolve, reject);
                    }catch(error) {
                        reject(x);
                    }
                })
            }
            if (this.status === REJECTED) {
                setTimeout(() => {
                    try{
                        let x = onRejected(this.reason);
                        resolvePromise(promise2, x, resolve, reject);
                    }catch(error) {
                        reject(x);
                    }
                })
            }
            if (this.status === PENGDING) {
                this.resolveCallback.push(() => {
                    setTimeout(() => {
                        try{
                            let x = onFulfilled(this.value);
                            resolvePromise(promise2, x, resolve, reject);
                        }catch(error) {
                            reject(error);
                        }
                    })
                });
                this.rejectCallback.push(() => {
                    setTimeout(() => {
                        try{
                            let x = onRejected(this.reason);
                            resolvePromise(promise2, x, resolve, reject);
                        }catch(error) {
                            reject(error);
                        }
                    })
                })
            }
        });
        return promise2;
    }
    catch(cb) {
        return this.then(null, cb);
    }
    static resolve(value) {
        return new Promise((resolve, reject) => {
            resolve(value);
        })
    }
    static reject(reason) {
        return new Promise((resolve, reject) => {
            reject(reason);
        })
    }
    static all(promises) {
        return new Promise((resolve, reject) => {
            let arr = [];
            let index = 0;
            let processData = (i, y) => {
                arr[i] = y;
                if (++index === promises.length) {
                    resolve(arr);
                }
            }
            for(let i = 0; i < promises.length; i++){
                let value = promise[i];
                if (isPromise(value)) {
                    value.then(y => {
                        processData(i, y);
                    }, reject)
                }else {
                    processData(i, value);
                }
            }
        })
    }
    static race(promises) {
        return new Promise((resolve, reject) => {
            promises.forEach(p => {
                if (isPromise(p)) {
                    p.then(resolve, reject)
                }else {
                    resolve(p);
                }
            })
        })
    }
    finally(cb) {
       return this.then(value => {
           return Promise.resolve(cb()).then(() => value);
       }, err => {
           return Promise.resolve(cb()).then(() => throw err);
       })
    }
    static try(cb) { // 错误需要在下一个then中捕获
        return new Promise((resolve, reject) => {
            resolve(cb());
        })
    }
}
    • wrap函数，阻止某一promise继续

    wrap包裹一个promise,并且给这个promise提供一个abort方法，可以让这个promise变成失败态

    function wrap(p1) {
    // 在内部再构建一个promise，将它的失败方法暴露到abort上
    // 如果调用了obort方法，会让这promise.race 立即失败
    let abort = null;
    let p = new Promise((resolve, reject) => {
        abort = reject;
    })
    let newPromise = Promise.race([p, p1]);
    newPromise.abort = abort;
    return newPromise;
}
let p = wrap(new Promise((resolve, reject) => {
    setTimeout(() => {
        resolve('成功'); // 如果超时，只是抛弃掉了执行结果而已
    }, 3000);
}))
// 如果超过2s，这个promise的成功结果我就不要了
p.then((data) => {
    console.log(data);
}, err => {
    console.log('err', err);
})
setTimeout(() => {
    p.abort('超时');
})
