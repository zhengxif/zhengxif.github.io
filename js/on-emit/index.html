    # 发布订阅与观察者模式

    发布订阅模式

    • 发布订阅的好处可以解耦合
    let fs = require('fs);
let school = {};
let event = {
    arr: [], // 中介
    on(fn) { // 订阅
        this.arr.push(arr); 
    },
    emit() { // 发布
        this.arr.forEach(fn => fn()); 
    }
}
event.on(function() {
    console.log('读取一次');
})
event.on(function() {
    if (Object.keys(school).length === 2) {
        console.log(school);
    }
})
fs.readFile('a.txt', 'utf8', function(err, data) {
    school['name'] = data;
    event.emit(); // 发布
})
fs.readFile('a.txt', 'utf8', function(err, data) {
    school['name'] = data;
    event.emit(); // 发布
})

    观察者模式

    • 观察者模式基于发布订阅，vue就是典型的观察者模式
    class Subject{ // 被观察者
    constructor() {
        this.arr = []; // 存放观察者的实例
        this.state = '正常';
    }
    attach(o) { // 挂载观察者
        this.arr.push(o);
    }
    setState(newState) {
        this.state = newState;
        this.arr.forEach( o => o.update(this));
    }
}
class Observer{ // 观察者
    constructor(name) {
        this.name = name;
    }
    update(s) {
        console.log(s.state + ':' + this.name);
    }
}
let s = new Subject();
let o1 = new Observer('张三');
let o2 = new Observer('李四');
s.attach(o1);
s.attach(o2);
s.setState('改变了');
