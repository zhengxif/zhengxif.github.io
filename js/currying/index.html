    # 柯里化

    何为柯里化？

    • 函数柯里化，是固定部分参数，返回一个接受剩余参数的函数，也称为部分计算函数，目的是为了缩小适用范围，创建一个针对性更强的函数。

    实现一个柯里化的通用函数

    function add(a, b, c, d, e) {
    return a + b + c + d + e;
}
function currying(fn, arr = []) {
    let len = fn.length;
    return function(...args) {
        arr = [...arr, ...args];
        if (arr.length < len) {
            // 参数数量不够时， 递归返回函数继续接受参数
            return currying(fn, arr);
        }else {
            return fn(...arr);
        }
    }
}
let res = currying(add)(1)(2,3)(4)(5);
console.log(res); // 15

    反柯里化实现

    • 反柯里化是为了让函数变得更通用些，范围更广,使本来只有特定对象才适用的方法，扩展到更多的对象。

    反柯里化通用函数实现

    function uncurring(fn) {
    // 保证调用的是原型上的apply
    // 1) 改变this 2) 让函数执行 3）把参数传入

    // 就是将fn执行，并且将后续的参数传递给 apply方法
    return function(context, ...args) {
        return Reflect.apply(fn, context, args);
    }
    // 相当于
    // return Function.prototype.apply.call(fn, context, args);   
}
let join = uncurring(Array.prototype.join);
// 内部基于call来实现的
let r = join([1,2,3],':');
console.log(r);
