    # 高阶函数

    何为高阶函数？

    1. 你可以将函数当作另外一个函数的参数传入
    2. 如果一个函数返回一个新的函数，那么这个函数也是高阶函数

    常见用途

    1. 可以用于批量生成函数
    let toString = Object.prototype.toString;
let isString = function(obj) {
    return toString.call(obj) === '[Object String]';
}
let isFunction = function(obj) {
    return toString.call(obj) === '[Object Function]';
}
let isType = function(type) {
    return function(obj) {
        return toString.call(obj) === `[Object ${type}]`;
    }
}
    1. 可以用于需要多次调用才执行的函数
    let after = function(time, callback) {
    return function() {
        if (--time == 0) {
            return callback.apply(this, arguments);
        }
    }
}
let fn = after(3, function() {
    console.log('3');
})
fn();
fn();
fn();
    1. 装饰模式
    // 每个函数在执行之前做一些其他事情 
Function.prototype.before = function(cb) {
    // 注意箭头函数中没有arguments
    // ...剩余运算符，收集参数成一个数组
    return (...args) => {
        cb();
        // 这里的this会向向一层作用域（上下文）中查找，即调用该before方法的函数
        this(...args);
    }
}
// 核型函数
function makeCoffee(who) {
    console.log(`创建一杯咖啡给：${who}`);
}
let newFun = makeCoffee.before(function() {
    console.log('加糖');
});
newFun('老板');

    比如react中的事务，在执行之前加入一些功能，在执行之后也加入一些功能。

    function perform(cb, arr) {
    return function() {
        arr.forEach(fn => fn.initialize());
        cb();
        arr.forEach(fn => fn.close());
    }
}
let newFunc = perform(function() {
    console.log('普通函数，核心功能');
}, [ //wrapper数组
    {
        initialize() {
           console.log('wrapper1 start');
        },
        close() {
            console.log('wrapper1 close');
        }
   },
   {
       initialize() {
           console.log('wrapper2 start');
       },
       close() {
           console.log('wrapper2 close');
       }
   }

]);
newFunc();
